19  import static org.assertj.core.api.Assertions.assertThatExceptionOfType;
20  
21  import org.assertj.core.api.SoftAssertions;
22  import org.assertj.core.api.junit.jupiter.SoftAssertionsExtension;
23  import org.bremersee.xml.adapter.DateXmlAdapter;
24  import org.bremersee.xml.model1.ObjectFactory;
25  import org.bremersee.xml.model1.Person;
26  import org.bremersee.xml.model6.StandaloneModel;
27  import org.junit.jupiter.api.Test;
28  import org.junit.jupiter.api.extension.ExtendWith;
29  
30  /**
31   * The jaxb context data test.
32   *
33   * @author Christian Bremer
34   */
35  @ExtendWith({SoftAssertionsExtension.class})
36  class JaxbContextDataTest {
37  
38    /**
39     * With class.
40     *
41     * @param softly the soft assertions
42     */
43    @Test
44    void withClass(SoftAssertions softly) {
45      JaxbContextData actual = new JaxbContextData(
46          StandaloneModel.class,
47          "http://localhost/standalone.xsd",
48          "http://localhost/standalone.xsd");
49      softly.assertThat(actual.getJaxbClasses())
50          .containsExactly(StandaloneModel.class);
51      softly.assertThat(actual.getNameSpacesWithSchemaLocations())
52          .containsExactly(new SchemaLocation(
53              "http://bremersee.org/xmlschemas/common-xml-test-model-6",
54              "http://localhost/standalone.xsd"));
55      softly.assertThat(actual.getKey())
56          .isEqualTo(StandaloneModel.class);
57      softly.assertThat(actual.toString())
58          .contains(StandaloneModel.class.getName());
59      softly.assertThat(actual.compareTo(new JaxbContextData(StandaloneModel.class)))
60          .isEqualTo(0);
61      softly.assertThat(actual)
62          .isNotEqualTo(new JaxbContextData(StandaloneModel.class, ""));
63    }
64  
65    /**
66     * With class as package.
67     *
68     * @param softly the soft assertions
69     */
70    @Test
71    void withClassAsPackage(SoftAssertions softly) {
72      JaxbContextData actual = new JaxbContextData(
73          ObjectFactory.class,
74          "http://bremersee.github.io/xmlschemas/common-xml-test-model-1.xsd");
75      softly.assertThat(actual.getJaxbClasses())
76          .containsExactly(Person.class);
77      softly.assertThat(actual.getNameSpacesWithSchemaLocations())
78          .containsExactly(new SchemaLocation(
79              "http://bremersee.org/xmlschemas/common-xml-test-model-1",
80              "http://bremersee.github.io/xmlschemas/common-xml-test-model-1.xsd"));
81    }
82  
83    /**
84     * With illegal class.
85     */
86    @Test
87    void withIllegalClass() {
88      assertThatExceptionOfType(IllegalArgumentException.class)
89          .isThrownBy(() -> new JaxbContextData(DateXmlAdapter.class));
90    }
91  
92    /**
93     * With illegal class as package.
94     */
95    @Test
96    void withIllegalClassAsPackage() {
97      assertThatExceptionOfType(IllegalArgumentException.class)
98          .isThrownBy(() -> new JaxbContextData(DateXmlAdapter.class, ""));
99    }
100 
101   /**
102    * With illegal package.
103    */
104   @Test
105   void withIllegalPackage() {
106     assertThatExceptionOfType(IllegalArgumentException.class)
107         .isThrownBy(() -> new JaxbContextData(DateXmlAdapter.class.getPackage()));
108   }
109 
110   /**
111    * With package.
112    *
113    * @param softly the soft assertions
114    */
115   @Test
116   void withPackage(SoftAssertions softly) {
117     JaxbContextData actual = new JaxbContextData(
118         ObjectFactory.class.getPackage(),
119         "http://bremersee.github.io/xmlschemas/common-xml-test-model-1.xsd");
120     softly.assertThat(actual.getKey())
121         .isEqualTo(ObjectFactory.class.getPackage());
122     softly.assertThat(actual.getJaxbClasses())
123         .containsExactly(Person.class);
124     softly.assertThat(actual.getNameSpacesWithSchemaLocations())
125         .containsExactly(new SchemaLocation(
126             "http://bremersee.org/xmlschemas/common-xml-test-model-1",
127             "http://bremersee.github.io/xmlschemas/common-xml-test-model-1.xsd"));
128     softly.assertThat(actual)
129         .isEqualTo(new JaxbContextData(ObjectFactory.class));
130     softly.assertThat(actual)
131         .isEqualTo(new JaxbContextData(ObjectFactory.class,
132             "http://bremersee.github.io/xmlschemas/common-xml-test-model-1.xsd"));
133     softly.assertThat(actual)
134         .isEqualTo(new JaxbContextData(ObjectFactory.class.getPackage()));
135   }
