47    /**
48     * The can marshal all function. If this function is set, the builder can marshal all objects
49     * which are annotated with {@code XmlRootElement}. This is the default behaviour.
50     *
51     * @see #withCanMarshal(BiFunction)
52     */
53    BiFunction<Class<?>, Map<String, JaxbContextData>, Boolean> CAN_MARSHAL_ALL
54        = (aClass, predefinedData) -> aClass != null
55        && aClass.isAnnotationPresent(XmlRootElement.class);
56  
57    /**
58     * The can unmarshal all function. If this function is set, the builder can unmarshal all objects
59     * which are annotated with {@code XmlRootElement}. This is the default behaviour.
60     *
61     * @see #withCanUnmarshal(BiFunction)
62     */
63    BiFunction<Class<?>, Map<String, JaxbContextData>, Boolean> CAN_UNMARSHAL_ALL
64        = (aClass, predefinedData) -> aClass != null
65        && aClass.isAnnotationPresent(XmlRootElement.class);
66  
67    /**
68     * The can marshal only predefined data function. If this function is set, the builder can marshal
69     * only objects which were added previously to the builder.
70     *
71     * @see #withCanMarshal(BiFunction)
72     */
73    BiFunction<Class<?>, Map<String, JaxbContextData>, Boolean> CAN_MARSHAL_ONLY_PREDEFINED_DATA
74        = (aClass, predefinedData) -> aClass != null
75        && aClass.isAnnotationPresent(XmlRootElement.class)
76        && predefinedData.containsKey(aClass.getPackage().getName());
77  
78    /**
79     * The can unmarshal only predefined data function. If this function is set, the builder can
80     * unmarshal only objects which were added previously to the builder.
81     *
82     * @see #withCanUnmarshal(BiFunction)
83     */
84    BiFunction<Class<?>, Map<String, JaxbContextData>, Boolean> CAN_UNMARSHAL_ONLY_PREDEFINED_DATA
85        = (aClass, predefinedData) -> aClass != null
86        && aClass.isAnnotationPresent(XmlRootElement.class)
87        && predefinedData.containsKey(aClass.getPackage().getName());

214   /**
215    * Add the given context path (package names which are separated by colon) to the jaxb context
216    * builder. This is the same as {@link javax.xml.bind.JAXBContext#newInstance(String)}.
217    *
218    * @param contextPath the context path (package names which are separated by colon)
219    * @return the jaxb context builder
220    */
221   default JaxbContextBuilder add(final String contextPath) {
222     if (StringUtils.hasText(contextPath)) {
223       final String[] packages = StringUtils.delimitedListToStringArray(contextPath, ":");
224       for (String pakkage : packages) {
225         add(new JaxbContextData(pakkage));
226       }
227     }
228     return this;
229   }

239   /**
240    * Add all jaxb context meta data to the jaxb context builder.
241    *
242    * @param data the data
243    * @return the jaxb context builder
244    */
245   default JaxbContextBuilder addAll(final Iterable<? extends JaxbContextData> data) {
246     return data == null ? this : addAll(data.iterator());
247   }
248 
249   /**
250    * Add all jaxb context meta data to the jaxb context builder.
251    *
252    * @param data the data
253    * @return the jaxb context builder
254    */
255   default JaxbContextBuilder addAll(final Iterator<? extends JaxbContextData> data) {
256     if (data != null) {
257       while (data.hasNext()) {
258         add(data.next());
259       }
260     }
261     return this;
262   }

264   /**
265    * Process the jaxb context meta data provider and add it's data to the jaxb context builder.
266    *
267    * @param dataProvider the data provider
268    * @return the jaxb context builder
269    */
270   default JaxbContextBuilder process(final JaxbContextDataProvider dataProvider) {
271     return dataProvider == null ? this : addAll(dataProvider.getJaxbContextData());
272   }
273 
274   /**
275    * Process the jaxb context meta data providers and add their data to the jaxb context builder.
276    *
277    * @param dataProviders the data providers
278    * @return the jaxb context builder
279    */
280   default JaxbContextBuilder processAll(
281       final Iterable<? extends JaxbContextDataProvider> dataProviders) {
282     return dataProviders == null ? this : processAll(dataProviders.iterator());
283   }
284 
285   /**
286    * Process the jaxb context meta data providers and add their data to the jaxb context builder.
287    *
288    * @param dataProviders the data providers
289    * @return the jaxb context builder
290    */
291   default JaxbContextBuilder processAll(
292       final Iterator<? extends JaxbContextDataProvider> dataProviders) {
293     if (dataProviders != null) {
294       while (dataProviders.hasNext()) {
295         process(dataProviders.next());
296       }
297     }
298     return this;
299   }
